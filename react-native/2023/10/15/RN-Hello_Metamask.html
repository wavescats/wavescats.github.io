  WalletConnectModal,
  useWalletConnectModal,
} from "@walletconnect/modal-react-native";

const providerMetadata = {
  name: "",
  description: "",
  url: "",

  redirect: {
    native: "",

  const { open, isConnected, address, provider } = useWalletConnectModal();

        providerMetadata={providerMetadata}
      />
    </View>
  );
}
